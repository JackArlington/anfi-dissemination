57 void AID::handleSelfMsg(cMessage* msg) {
58  switch (msg->getKind()) {
59 
60  case BROADCAST_TIMEOUT: {
61  MessageInfoEntry* info = (MessageInfoEntry*) msg->getContextPointer();
62 
63  if (isMessageAlive(info) && isInsideROI(info)) {
64  if (info->c > 1) {
65  int s = 0;
66  simtime_t interArrivalTimeBase = (info->broadcastTimer->getArrivalTime() - info->broadcastTimer->getCreationTime()) / info->c;
67 
68  for (std::vector<simtime_t>::iterator i = info->deltaTimes.begin(); i != info->deltaTimes.end(); i++) {
69  if (interArrivalTimeBase - (*i) > 0) {
70  s--;
71  } else {
72  s++;
73  }
74  }
75  if (s > 0) {
76  addMessageToOutputQueue(info);
77  }
78  } else {
79  addMessageToOutputQueue(info);
80  }
81  }
82 
83  delete info->broadcastTimer;
84  info->broadcastTimer = NULL;
85 
86  break;
87  }
88 
89  case SEND_DATA: {
90  sendData();
91 
92  break;
93  }
94 
95  default: {
96  if (msg)
97  EV << "AID - Error: Got Self Message of unknown kind! Name: " << msg->getName() << endl;
98  break;
99  }
100  }
101 }

105 void AID::onData(WaveShortMessage* wsm) {
106  DataMessage* dataMsg = dynamic_cast<DataMessage*>(wsm->decapsulate());
107 
108  MessageInfoEntry* info = new MessageInfoEntry;
109  info->messageID = wsm->getSerial();
110  info->messageOriginPosition = dataMsg->getMessageOriginPosition();
111  info->messageROI = dataMsg->getMessageROI();
112  info->messageOriginTime = dataMsg->getMessageOriginTime();
113  info->messageTTL = dataMsg->getMessageTTL();
114  info->hops = dataMsg->getHops() + 1;
115  info->receptionTime = simTime();
116  info->messageLength = wsm->getByteLength();
117  info->c = 1;
118  info->lastReceptionTime = simTime();
119  info->broadcastTimer = NULL;
120 
121  if (!isMessageAlive(info) || !isInsideROI(info)) {
122  delete info;
123  delete dataMsg;
124 
125  return;
126  }
127 
128  if (!isDuplicateMsg(info->messageID)) {
129  double delay = uniform(0, par("randomDelayTimeMax").doubleValue());
130 
131  info->broadcastTimer = new cMessage("broadcast suppression", BROADCAST_TIMEOUT);
132  info->broadcastTimer->setContextPointer((MessageInfoEntry*) info);
133  scheduleAt(simTime() + delay, info->broadcastTimer);
134 
135  messagesRcvd[info->messageID] = info;
136  emit(messagesReceived, 1);
137  } else {
138  delete info;
139  MessageInfoEntry* info = messagesRcvd[wsm->getSerial()];
140 
141  if (info->broadcastTimer && info->broadcastTimer->isScheduled()) {
142  info->c += 1;
143  info->deltaTimes.push_back(simTime() - info->lastReceptionTime);
144  info->lastReceptionTime = simTime();
145  }
146 
147  emit(duplicatedMessages, 1);
148  }
149 
150  delete dataMsg;
151 }
