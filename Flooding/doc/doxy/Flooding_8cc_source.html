74 void Flooding::onData(WaveShortMessage* wsm) {
75  MessageInfoEntry* info = extractMsgInfo(wsm);
76 
77  if (!isInsideROI(info) || !isMessageAlive(info)) {
78  return;

91 Flooding::MessageInfoEntry* Flooding::extractMsgInfo(WaveShortMessage* wsm) {

103  info->messageTTL = dataMsg->getMessageTTL();
104  info->hops = dataMsg->getHops() + 1;

122 void Flooding::sendData() {
123  if (!outputQueue.empty()) {
124  int nextMessageID = outputQueue.front();
125  outputQueue.pop_front();
126  MessageInfoEntry* info = messagesRcvd[nextMessageID];
127 
128  WaveShortMessage* wsm = createDataMsg(info);
129  sendWSM(wsm);
130 
131  emit(messagesTransmitted, 1);
132 
133  double nextPktTime = info->messageLength / datarate;
134  scheduleAt(simTime() + nextPktTime, dataRateTimer);
135  }
136 }
